#include <stdio.h>
#include <string.h> 

int main() {
    char str[10];
    printf("Enter the name: ");
    scanf("%s",&str);
    int length = strlen(str);
    int i, j;
    char temp;

    
    for (i = 0, j = length - 1; i < j; i++, j--) {
        temp = str[i];
        str[i] = str[j];
        str[j] = temp;
    }

    printf("Reversed string: %s\n", str); 

    return 0;
}



OUTPUT

Enter the name: KARTHI
Reversed string: IHTRAK
