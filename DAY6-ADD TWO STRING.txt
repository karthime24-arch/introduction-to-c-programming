#include <stdio.h>
int main() {
    char str1[100]; 
    char str2[50];

    printf("Enter the first string: ");
    scanf("%s", str1); 
    printf("Enter the second string: ");
    scanf("%s", str2); 

    int i = 0;
    while (str1[i] != '\0') {
        i++;
    }

    int j = 0;
    
    while (str2[j] != '\0') {
        str1[i] = str2[j];
        i++;
        j++;
    }
    str1[i] = '\0';

    printf("ADD TWO string: %s\n", str1);

    return 0;
}



OUTPUT

Enter the first string: KAR
Enter the second string: THI
ADD TWO string: KARTHI
