#include <stdio.h>
int main() {
    char str[100]; 
    int freq[256] = {0};
    int i = 0; 
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin); 
    while (str[i] != '\0') {
        
        if (str[i] == '\n') {
            i++;
            continue;
        }
        freq[str[i]]++;
        i++;
    }

  
    printf("Character frequencies:\n");
    for (i = 0; i < 256; i++) {
        if (freq[i] != 0) {
            printf("'%c' : %d\n", (char)i, freq[i]);
        }
    }

    return 0;
}


OUTPUT

Character frequencies:

'A' : 1
'H' : 1
'I' : 1
'K' : 1
'R' : 1
'S' : 1
'T' : 1
