#include<stdio.h>
int main() {
    int i,j,n;
    printf("Enter a number: ");
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        for(j=1;j<=5;j++){
            printf("i=%d,j=%d\n",i,j);
        }
      
}
return 0;
}
