#include <stdio.h>
#include <string.h>

void removeSpaces(char *str) {
    int count = 0; 
    int i;
    for (i = 0; str[i] != '\0'; i++) {
        if (str[i] != ' ') {
            str[count++] = str[i];
        }
    }
    str[count] = '\0';
}
int main() {
    char str[100];
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin); 
    str[strcspn(str, "\n")] = '\0';
    printf("Original string: \"%s\"\n", str);
    removeSpaces(str);
    printf("String after removing spaces: \"%s\"\n", str);

    return 0;
}



OUTPUT

Enter a string: ABCD EFGH IJKL MNOP QRST UVW XYZ
Original string: "ABCD EFGH IJKL MNOP QRST UVW XYZ"
String after removing spaces: "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
