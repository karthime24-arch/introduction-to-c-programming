#include <stdio.h>

int main() {
    int arr[100],i,n,x,count=0;
    printf("Enter the elements:");
    scanf("%d",&n);
    printf("Enter %d elements:\n",n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    printf("Enter element to count:\n");
    scanf("%d",&x);
    for(i=0;i<n;i++)
    {
        if(arr[i]==x)
        {
            count++;
        }
    }
    printf("%d occurs %d times\n",x,count);
    return 0;
}
