#include <stdio.h>

int main() {
    int A[100],B[100],n,i;
    printf("Enter number of elements:");
    scanf("%d",&n);
    printf("Enter %d elements of array A:\n",n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&A[i]);
    }
    for(i=0;i<n;i++)
    {
        B[i]=A[i];
    }
    printf("Array B after copying:\n");
    for(i=0;i<n;i++)
    {
        printf("%d",B[i]);
    }
    return 0;
}

OUTPUT

Enter number of elements:3
Enter 3 elements of array A:
1
2
3
Array B after copying:
123
