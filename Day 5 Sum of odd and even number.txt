#include <stdio.h>

int main() {
    int arr[100],n,i,evenSum=0,oddSum=0;
    printf("Enter number of elements:");
    scanf("%d",&n);
    printf("Enter %d elements:\n",n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    for(i=0;i<n;i++)
    {
        if(arr[i]%2==0){
            evenSum+=arr[i];
            
        }
        else{
            oddSum += arr[i];
        }
    }
    printf("Sum of Even numbers =%d\n",evenSum);
    printf("Sum of Odd numbers =%d\n",oddSum);
    return 0;

}
