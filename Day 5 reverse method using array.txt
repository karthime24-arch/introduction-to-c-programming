#include <stdio.h>

int main() {
    int arr[100],n,i;
    printf("Enter the elements:");
    scanf("%d",&n);
    printf("Enter %d elements:\n",n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    printf("Array in reverse order:\n");
    for(i=n-1;i>=0;i--)
    {
        printf("%d ",arr[i]);
    }
    printf("\n");
    return 0;
}

Enter 5 elements:
3
4
