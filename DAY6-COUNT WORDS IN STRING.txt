#include <stdio.h>
#include <string.h> 
int main() {
    char str[100]; 
    int i, totalwords;

    totalwords = 1; 

    printf("Enter any String : ");
    fgets(str, sizeof(str), stdin); 
    str[strcspn(str, "\n")] = 0; 

    for (i = 0; str[i] != '\0'; i++) {
        if (str[i] == ' ' || str[i] == '\n' || str[i] == '\t') {
            totalwords++;
        }
    }

    printf("The Total Number of Words in this String \"%s\" = %d\n", str, totalwords);

    return 0;
}



OUTPUT

Enter any String : ABCD EFGH IJKL MNOP QRST UVW XYZ
The Total Number of Words in this String "ABCD EFGH IJKL MNOP QRST UVW XYZ" = 7
