#include <stdio.h>

int main() {
    int a[2][2]={{1,2},{3,4}};
    int b[2][2]={{4.5},{6,7}};
    int c[2][2],i,j;
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            c[i][j]=a[i][j]+b[i][j];
            printf("%d",c[i][j]);
        }
        printf(" \n");
    }
  
    
    return 0;
}


OUTPUT

5  7
9  11
