#include<stdio.h>
int main() {
    int i,j,n;
    printf("Enter a number: ");
    scanf("%d",&n);
    for(i=n;i<=n;i--){
        for(j=1;j<=i;j++){
            printf("%d",j);
            printf(" ");
        }
        printf("\n");
}
return 0;
}
