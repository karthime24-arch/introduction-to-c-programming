#include <stdio.h>
#include <ctype.h> 
#include <string.h> 
int main() {
    char str[100];
    int i;

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    str[strcspn(str, "\n")] = 0; 
    char upperStr[100];
    strcpy(upperStr, str);
    for (i = 0; upperStr[i] != '\0'; i++) {
        upperStr[i] = toupper(upperStr[i]);
    }
    printf("Uppercase string: %s\n", upperStr);
    char lowerStr[100];
    strcpy(lowerStr, str);
    for (i = 0; lowerStr[i] != '\0'; i++) {
        lowerStr[i] = tolower(lowerStr[i]);
    }
    printf("Lowercase string: %s\n", lowerStr);
    printf("\nChecking case of individual characters:\n");
    for (i = 0; str[i] != '\0'; i++) {
        if (isalpha(str[i])) { 
            if (isupper(str[i])) {
                printf("'%c' is an uppercase character.\n", str[i]);
            } else if (islower(str[i])) {
                printf("'%c' is a lowercase character.\n", str[i]);
            }
        } else {
            printf("'%c' is not an alphabetic character.\n", str[i]);
        }
    }

    return 0;
}





OUTPUT


Enter a string: KARTHI S
Uppercase string: KARTHI S
Lowercase string: karthi s

Checking case of individual characters:
'K' is an uppercase character.
'A' is an uppercase character.
'R' is an uppercase character.
'T' is an uppercase character.
'H' is an uppercase character.
'I' is an uppercase character.
' ' is not an alphabetic character.
'S' is an uppercase character.
