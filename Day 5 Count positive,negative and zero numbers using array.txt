#include <stdio.h>

int main() {
    int arr[] = {0, -5, 3, 9, -2, 0, 7, -8};
    int n = sizeof(arr) / sizeof(arr[0]);
    int positive = 0, negative = 0, zero = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0)
            positive++;
        else if (arr[i] < 0)
            negative++;
        else
            zero++;
    }
    printf("Positive numbers: %d\n", positive);
    printf("Negative numbers: %d\n", negative);
    printf("Zeroes: %d\n", zero);
    return 0;
}
